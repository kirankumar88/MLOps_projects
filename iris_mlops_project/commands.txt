# Git Commands for Iris MLOps Project

# Step 1: Navigate to project and initialize git
cd /Users/pudurlakshmivenkateswarlu/codes/MLOPS/iris_mlops_project
git init

# Step 2: Add all files and check status
git add .
git status

# Step 3: Create initial commit
git commit -m "Initial commit: Iris MLOps project

- Data loading and preprocessing pipeline
- Feature engineering with scaling and polynomial features
- Model training (Logistic Regression, Random Forest)
- Model evaluation with metrics and visualizations
- Model registry for versioning
- Configuration management
- Logging setup

# Step 4: Add remote (GitHub)
git remote add origin https://github.com/VenkateswarluPudur/mlops.git

# Step 5: Push to GitHub
git push -u origin main

# =============================================================================
# GIT BRANCH WORKFLOW - Create Branch, Push, and Merge
# =============================================================================

# Step 1: Make sure you're on main and up to date
git checkout main
git pull origin main

# Step 2: Create and switch to a new branch
git checkout -b feature/new-feature
# Examples:
# git checkout -b feature/add-api
# git checkout -b bugfix/fix-login
# git checkout -b hotfix/security-patch

# Step 3: Make your changes, then stage and commit
git add .
git status
git commit -m "Add new feature description"

# Step 4: Push the new branch to remote
git push -u origin feature/new-feature

# =============================================================================
# MERGING OPTIONS
# =============================================================================

# OPTION A: Merge locally (for solo projects)
git checkout main
git merge feature/new-feature
git push origin main

# OPTION B: Create Pull Request on GitHub (recommended for teams)
# 1. Go to: https://github.com/VenkateswarluPudur/mlops/pulls
# 2. Click "New Pull Request"
# 3. Select your branch (feature/new-feature)
# 4. Add title and description
# 5. Click "Create Pull Request"
# 6. Review and click "Merge Pull Request"

# =============================================================================
# CLEANUP AFTER MERGE
# =============================================================================

# Delete local branch
git branch -d feature/new-feature

# Delete remote branch
git push origin --delete feature/new-feature

# Verify branches
git branch -a

# =============================================================================
# USEFUL BRANCH COMMANDS
# =============================================================================

# List all branches
git branch -a

# Switch to existing branch
git checkout branch-name

# See current branch
git branch

# See branch differences
git diff main..feature/new-feature

# Fetch all remote branches
git fetch --all
